# import data
data_raw = read.table("Final Data.txt", header = TRUE, sep = "\t")

# DXXOFBMD = Total femur bone mineral density (gm/cm^2)
# DXXOSBMD = Total spine BMD

# remove rows missing femur or spine BMD
data = data_raw[complete.cases(data_raw[, c("DXXOFBMD", "DXXOSBMD")]), ]

# look at distribution of response variables
hist(data$DXXOFBMD)

hist(data$DXXOSBMD)

# create response variable which is average of femur and spine BMD
data$avgBMD = (data$DXXOFBMD + data$DXXOSBMD)/2

# Remove rows where any column starts with "DXX"
data_clean <- data[, !grepl("^DXX", names(data))]

# Calculate the percentage of missing data in each column
missing_percentage <- colSums(is.na(data_clean)) / nrow(data_clean) * 100

# Remove columns with large percentage missing data
data_clean <- data_clean[, missing_percentage <= 30]

# remove rows with NA
data_clean <- na.omit(data_clean)

# Check for columns with the same value in all rows
data_clean <- data_clean[, apply(data_clean, 2, function(x) length(unique(x)) > 1)]

# fit full model
model2 <- lm(avgBMD ~ ., data = data_clean)
